A Basic Command 

use crate::{Context, Error, utils::bot};

#[poise::command(slash_command)]
pub async fn (
    ctx: Context<'_>,
    #[description = "Send the response directly to you?"] direct: Option<bool>,
) -> Result<(), Error> {
    let direct = bot::defer_based_on_direct(ctx, direct).await?;

    Ok(())
}
